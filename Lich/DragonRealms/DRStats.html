def self.native_mana
  case DRStats.guild
  when 'Necromancer'
    'arcane'
  when 'Barbarian', 'Thief'
    nil
  when 'Moon Mage', 'Trader'
    'lunar'
  when 'Warrior Mage', 'Bard'
    'elemental'
  when 'Cleric', 'Paladin'
    'holy'
  when 'Empath', 'Ranger'
    'life'
  end
end
