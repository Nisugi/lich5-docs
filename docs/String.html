def split_as_list
  string = self
  string.sub!(/^You (?:also see|notice) |^In the .+ you see /, ',')
  string.sub('.', '').sub(/ and (an?|some|the)/, ', \1').split(',').reject { |str| str.strip.empty? }.collect { |str| str.lstrip }
end
