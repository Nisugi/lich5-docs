def expiration(effect)
  if effect.is_a?(Regexp)
    to_h.find { |k, _v| k.to_s =~ effect }[1] || 0
  else
    to_h.fetch(effect, 0)
  end
end

def time_left(effect)
  if expiration(effect) != 0
    ((expiration(effect) - Time.now) / 60.to_f)
  else
    expiration(effect)
  end
end
