def all_wounds
  fix_injury_mode('both')
  XMLData.injuries.transform_values { |v| v['wound'] }
end

def arms

def wound_level(part)
  fix_injury_mode('both')
  XMLData.injuries[part.to_s] && XMLData.injuries[part.to_s]['wound']
end
