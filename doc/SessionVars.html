Module: SessionVars

Defined in:

Overview

A module for managing session-level variables that need to be shared between scripts but don’t require persistent storage. Provides a hash-like interface for storing and retrieving temporary variables during runtime.

def SessionVars.[]=(name, val)
  if val.nil?
    @@svars.delete(name)
  else
    @@svars[name] = val
  end
end

def SessionVars.method_missing(arg1, arg2 = '')
  if arg1[-1, 1] == '='
    if arg2.nil?
      @@svars.delete(arg1.to_s.chop)
    else
      @@svars[arg1.to_s.chop] = arg2
    end
  else
    @@svars[arg1.to_s]
  end
end
